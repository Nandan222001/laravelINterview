1. What are PHP 8 attributes and how do they differ from annotations in docblocks?

Intermediate attributes

PHP 8 attributes are a language feature for adding metadata to classes, properties, methods, and parameters. They differ from docblock annotations in that they are processed at runtime via the Reflection API with full type safety and compile-time validation. Docblock annotations are just comments and require custom parsing.

5. Create a #[Cached] attribute that works with a caching decorator pattern.

Advanced attributes

Create a #[Cached] attribute using a decorator pattern:

#[Attribute(Attribute::TARGET_METHOD)]
class Cached
{
    public function __construct(
        public int $ttl = 3600,
        public string $key = ''
    ) {}
}

// Decorator implementation
class CacheDecorator
{
    public function __call($method, $args)
    {
        $refMethod = new ReflectionMethod($this->wrapped, $method);
        $attrs = $refMethod->getAttributes(Cached::class);
        
        if (!empty($attrs)) {
            $cached = $attrs[0]->newInstance();
            $cacheKey = $cached->key ?: get_class($this->wrapped) . ":{$method}";
            
            return Cache::remember($cacheKey, $cached->ttl, function() use ($method, $args) {
                return $this->wrapped->$method(...$args);
            });
        }
        
        return $this->wrapped->$method(...$args);
    }
}

21. Explain the difference between backed and pure enums in PHP 8.1.

Intermediate enums

Backed enums have an underlying scalar value (int or string) accessible via ->value, while pure enums have no value.

// Backed enum
enum Status: string {
    case Active = 'active';
    case Inactive = 'inactive';
}

// Pure enum
enum Color {
    case Red;
    case Blue;
}

41. Explain how PHP 8's JIT compiler works and when it's beneficial.

Expert jit

PHP 8's JIT (Just-In-Time) compiler converts opcodes to native machine code at runtime, speeding up CPU-intensive code. Beneficial for computational tasks (math, loops); has minimal impact on typical I/O-bound web applications.

# Enable JIT in php.ini
opcache.jit=tracing
opcache.jit_buffer_size=100M

Best used for batch processing, data analysis, and mathematical computations rather than typical web request handlers.

131. Write a custom middleware that logs request/response timing.

Advanced middleware

Custom middleware for logging request/response timing:

namespace App\Http\Middleware;

use Closure;
use Illuminate\Http\Request;
use Illuminate\Support\Facades\Log;

class RequestTimingMiddleware
{
    public function handle(Request $request, Closure $next)
    {
        $startTime = microtime(true);
        
        $response = $next($request);
        
        $duration = microtime(true) - $startTime;
        
        Log::info('Request timing', [
            'method' => $request->method(),
            'uri' => $request->getRequestUri(),
            'duration' => round($duration * 1000, 2) . 'ms',
            'status' => $response->status()
        ]);
        
        return $response;
    }
}

301. Write a complete Razorpay payment integration service.

Complete Razorpay payment integration service:

namespace App\Services;

use Razorpay\Api\Api;
use Exception;

class RazorpayService
{
    protected $api;
    
    public function __construct()
    {
        $this->api = new Api(
            config('services.razorpay.key'),
            config('services.razorpay.secret')
        );
    }
    
    public function createOrder(array $data)
    {
        try {
            return $this->api->order->create([
                'amount' => $data['amount'] * 100, // Convert to paisa
                'currency' => $data['currency'] ?? 'INR',
                'receipt' => $data['receipt'],
                'payment_capture' => 1
            ]);
        } catch (Exception $e) {
            logger()->error('Razorpay order creation failed', [
                'error' => $e->getMessage(),
                'data' => $data
            ]);
            throw $e;
        }
    }
    
    public function verifyPayment(array $attributes)
    {
        $signature = hash_hmac(
            'sha256',
            $attributes['razorpay_order_id'] . '|' . $attributes['razorpay_payment_id'],
            config('services.razorpay.secret')
        );
        
        return $signature === $attributes['razorpay_signature'];
    }
    
    public function capturePayment(string $paymentId, int $amount)
    {
        return $this->api->payment->fetch($paymentId)->capture([
            'amount' => $amount
        ]);
    }
}

492. Write code to prevent SQL injection in Laravel.

Expert security

Laravel's query builder and Eloquent ORM prevent SQL injection through parameterized queries:

// Safe: Using query builder with parameter binding
$users = DB::table('users')
    ->where('email', $email)
    ->get();

// Safe: Using Eloquent
$user = User::where('status', $status)->first();

// Safe: Raw queries with bindings
$users = DB::select('SELECT * FROM users WHERE email = ?', [$email]);

// UNSAFE: Raw SQL without bindings (Never do this!)
// $users = DB::select("SELECT * FROM users WHERE email = '$email'");

// Safe: Using named bindings
$users = DB::select('SELECT * FROM users WHERE email = :email', [
    'email' => $email
]);

Always use parameter binding for user input. Never concatenate user input into SQL queries.

641. Write a payment processing queue job.

Advanced payment-integration

Payment processing queue job implementation:

namespace App\Jobs;

use App\Models\Payment;
use App\Services\PaymentGateway;
use Illuminate\Bus\Queueable;
use Illuminate\Contracts\Queue\ShouldQueue;
use Illuminate\Foundation\Bus\Dispatchable;
use Illuminate\Queue\InteractsWithQueue;
use Illuminate\Queue\SerializesModels;

class ProcessPayment implements ShouldQueue
{
    use Dispatchable, InteractsWithQueue, Queueable, SerializesModels;

    public $tries = 3;
    public $timeout = 120;
    public $backoff = [60, 120, 300];

    protected $payment;

    public function __construct(Payment $payment)
    {
        $this->payment = $payment;
    }

    public function handle(PaymentGateway $gateway)
    {
        try {
            $result = $gateway->processPayment([
                'amount' => $this->payment->amount,
                'currency' => $this->payment->currency,
                'payment_method' => $this->payment->payment_method_id
            ]);

            $this->payment->update([
                'status' => 'completed',
                'transaction_id' => $result->id,
                'completed_at' => now()
            ]);

        } catch (\Exception $e) {
            $this->payment->update([
                'status' => 'failed',
                'error_message' => $e->getMessage()
            ]);

            throw $e;
        }
    }

    public function failed(\Throwable $exception)
    {
        // Notify admin, log, update payment status
        logger()->error('Payment processing failed', [
            'payment_id' => $this->payment->id,
            'error' => $exception->getMessage()
        ]);
    }
}

541. Write a token bucket rate limiter implementation.

Intermediate security

Token bucket rate limiter implementation:

namespace App\Services;

use Illuminate\Support\Facades\Redis;

class TokenBucketRateLimiter
{
    protected $capacity;
    protected $refillRate;
    protected $prefix = 'rate_limit:';

    public function __construct(int $capacity = 100, int $refillRate = 10)
    {
        $this->capacity = $capacity;
        $this->refillRate = $refillRate;
    }

    public function allow(string $key): bool
    {
        $tokenKey = $this->prefix . $key . ':tokens';
        $timestampKey = $this->prefix . $key . ':timestamp';

        $now = microtime(true);
        $lastRefill = Redis::get($timestampKey) ?? $now;
        $tokens = Redis::get($tokenKey) ?? $this->capacity;

        // Calculate tokens to add based on time elapsed
        $elapsed = $now - $lastRefill;
        $tokensToAdd = $elapsed * $this->refillRate;
        $tokens = min($this->capacity, $tokens + $tokensToAdd);

        if ($tokens >= 1) {
            $tokens -= 1;
            Redis::set($tokenKey, $tokens);
            Redis::set($timestampKey, $now);
            return true;
        }

        return false;
    }

    public function remaining(string $key): int
    {
        return (int) Redis::get($this->prefix . $key . ':tokens') ?? $this->capacity;
    }
}

801. Write AES-256 encryption service in Laravel.

Expert security

AES-256 encryption service implementation:

namespace App\Services;

class EncryptionService
{
    protected $cipher = 'AES-256-CBC';
    protected $key;

    public function __construct()
    {
        $this->key = config('app.key');
    }

    public function encrypt(string $data): string
    {
        $ivLength = openssl_cipher_iv_length($this->cipher);
        $iv = openssl_random_pseudo_bytes($ivLength);
        
        $encrypted = openssl_encrypt(
            $data,
            $this->cipher,
            $this->key,
            OPENSSL_RAW_DATA,
            $iv
        );
        
        // Combine IV and encrypted data
        $result = base64_encode($iv . $encrypted);
        
        // Add HMAC for integrity verification
        $hmac = hash_hmac('sha256', $result, $this->key);
        
        return base64_encode($hmac . $result);
    }

    public function decrypt(string $data): string
    {
        $data = base64_decode($data);
        
        // Extract and verify HMAC
        $hmac = substr($data, 0, 64);
        $data = substr($data, 64);
        
        $expectedHmac = hash_hmac('sha256', $data, $this->key);
        
        if (!hash_equals($hmac, $expectedHmac)) {
            throw new \Exception('HMAC verification failed');
        }
        
        $data = base64_decode($data);
        $ivLength = openssl_cipher_iv_length($this->cipher);
        
        $iv = substr($data, 0, $ivLength);
        $encrypted = substr($data, $ivLength);
        
        return openssl_decrypt(
            $encrypted,
            $this->cipher,
            $this->key,
            OPENSSL_RAW_DATA,
            $iv
        );
    }
}

901. Write a multi-layer caching strategy.

Advanced middleware

Multi-layer caching strategy implementation:

namespace App\Services;

use Illuminate\Support\Facades\Cache;
use Illuminate\Support\Facades\Redis;

class MultiLayerCache
{
    protected $l1Cache = []; // Memory cache
    protected $l2CacheKey = 'l2:';
    protected $l3CacheKey = 'l3:';

    public function get(string $key, callable $callback, int $ttl = 3600)
    {
        // Layer 1: In-memory cache (fastest)
        if (isset($this->l1Cache[$key])) {
            return $this->l1Cache[$key];
        }

        // Layer 2: Redis cache (fast)
        $l2Value = Redis::get($this->l2CacheKey . $key);
        if ($l2Value !== null) {
            $this->l1Cache[$key] = unserialize($l2Value);
            return $this->l1Cache[$key];
        }

        // Layer 3: File/Database cache (slower)
        $l3Value = Cache::get($this->l3CacheKey . $key);
        if ($l3Value !== null) {
            Redis::setex($this->l2CacheKey . $key, 300, serialize($l3Value));
            $this->l1Cache[$key] = $l3Value;
            return $l3Value;
        }

        // Cache miss - execute callback and populate all layers
        $value = $callback();
        
        $this->l1Cache[$key] = $value;
        Redis::setex($this->l2CacheKey . $key, 300, serialize($value));
        Cache::put($this->l3CacheKey . $key, $value, $ttl);

        return $value;
    }

    public function invalidate(string $key): void
    {
        unset($this->l1Cache[$key]);
        Redis::del($this->l2CacheKey . $key);
        Cache::forget($this->l3CacheKey . $key);
    }
}
