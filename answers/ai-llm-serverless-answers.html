1. LLM Fundamentals & Integration

What are Large Language Models (LLMs)?

Large Language Models are neural networks trained on massive text datasets. They learn patterns in language and can generate text, answer questions, translate, and summarize. Examples include GPT-4, Claude, and Gemini.

OpenAI API Integration

import openai
import os
from typing import List, Dict

openai.api_key = os.environ['OPENAI_API_KEY']

class OpenAIClient:
    def __init__(self, model: str = "gpt-4-turbo-preview"):
        self.model = model
    
    def chat_completion(self, messages: List[Dict], 
                       temperature: float = 0.7,
                       max_tokens: int = None) -> Dict:
        """Basic chat completion"""
        response = openai.ChatCompletion.create(
            model=self.model,
            messages=messages,
            temperature=temperature,
            max_tokens=max_tokens
        )
        
        return {
            'content': response.choices[0].message.content,
            'usage': {
                'prompt_tokens': response.usage.prompt_tokens,
                'completion_tokens': response.usage.completion_tokens,
                'total_tokens': response.usage.total_tokens
            }
        }
    
    def stream_completion(self, messages: List[Dict]):
        """Streaming response for better UX"""
        response = openai.ChatCompletion.create(
            model=self.model,
            messages=messages,
            stream=True
        )
        
        for chunk in response:
            if chunk.choices[0].delta.get('content'):
                yield chunk.choices[0].delta.content

# Usage
client = OpenAIClient()
messages = [
    {"role": "system", "content": "You are a helpful assistant."},
    {"role": "user", "content": "Explain RAG in one paragraph."}
]

result = client.chat_completion(messages)
print(result['content'])
print(f"Tokens used: {result['usage']['total_tokens']}")

                      system: str = None,
                      max_tokens: int = 1024) -> Dict:

            'max_tokens': max_tokens,
            'messages': messages
        }

        return {
            'content': response.content[0].text,
            'usage': {
                'input_tokens': response.usage.input_tokens,
                'output_tokens': response.usage.output_tokens
            }
        }

3. Embeddings & Vector Databases

What are Embeddings?

Embeddings convert text to vector representations. Similar texts have similar embeddings. Used for similarity search, clustering, and recommendations.

Generating Embeddings with OpenAI

import openai

def generate_embeddings(texts: List[str]) -> List[List[float]]:
    """Generate embeddings for texts"""
    response = openai.Embedding.create(
        input=texts,
        model="text-embedding-3-small"  # or text-embedding-ada-002
    )
    
    return [item['embedding'] for item in response['data']]

# Usage
texts = [
    "What are the loan requirements?",
    "How to apply for a mortgage?",
    "Personal loan eligibility criteria"
]

embeddings = generate_embeddings(texts)
print(f"Embedding dimension: {len(embeddings[0])}")  # 1536 for ada-002

Pinecone Vector Database

import pinecone
import openai

# Initialize Pinecone
pinecone.init(
    api_key=os.environ['PINECONE_API_KEY'],
    environment=os.environ['PINECONE_ENVIRONMENT']
)

class PineconeVectorDB:
    def __init__(self, index_name: str):
        # Create index if doesn't exist
        if index_name not in pinecone.list_indexes():
            pinecone.create_index(
                name=index_name,
                dimension=1536,  # ada-002 dimension
                metric='cosine'
            )
        
        self.index = pinecone.Index(index_name)
    
    def upsert_documents(self, documents: List[Dict]):
        """Insert/update documents with embeddings"""
        # Generate embeddings
        texts = [doc['text'] for doc in documents]
        embeddings = generate_embeddings(texts)
        
        # Prepare vectors
        vectors = []
        for doc, embedding in zip(documents, embeddings):
            vectors.append({
                'id': doc['id'],
                'values': embedding,
                'metadata': {
                    'text': doc['text'],
                    **doc.get('metadata', {})
                }
            })
        
        # Upsert to Pinecone
        self.index.upsert(vectors=vectors)
    
    def search(self, query: str, top_k: int = 5, 
               filter: Dict = None) -> List[Dict]:
        """Search similar documents"""
        # Generate query embedding
        query_embedding = generate_embeddings([query])[0]
        
        # Search
        results = self.index.query(
            vector=query_embedding,
            top_k=top_k,
            filter=filter,
            include_metadata=True
        )
        
        # Format results
        matches = []
        for match in results['matches']:
            matches.append({
                'score': match['score'],
                'text': match['metadata']['text'],
                'metadata': match['metadata']
            })
        
        return matches

4. RAG (Retrieval-Augmented Generation) Systems

What is RAG?

RAG retrieves relevant documents from a knowledge base and provides them as context to an LLM. This reduces hallucinations and grounds responses in factual data.

RAG Pipeline

class RAGPipeline:
    """Complete RAG implementation"""
    
    def __init__(self, index_name: str):
        self.vector_db = PineconeVectorDB(index_name)
        self.llm_client = OpenAIClient()
    
    def ingest_documents(self, documents: List[Dict]):
        """Step 1-4: Ingest, chunk, embed, store"""
        chunks = []
        
        for doc in documents:
            # Split into chunks
            text_chunks = self.chunk_text(doc['content'])
            
            for i, chunk in enumerate(text_chunks):
                chunks.append({
                    'id': f"{doc['id']}_chunk_{i}",
                    'text': chunk,
                    'metadata': {
                        'source': doc['source'],
                        'chunk_index': i,
                        **doc.get('metadata', {})
                    }
                })
        
        # Embed and store
        self.vector_db.upsert_documents(chunks)
    
    def chunk_text(self, text: str, chunk_size: int = 1000, 
                   overlap: int = 200) -> List[str]:
        """Split text into overlapping chunks"""
        chunks = []
        start = 0
        
        while start < len(text):
            end = start + chunk_size
            chunk = text[start:end]
            chunks.append(chunk)
            start = end - overlap
        
        return chunks
    
    def retrieve(self, query: str, top_k: int = 5) -> List[Dict]:
        """Step 5: Retrieve relevant documents"""
        return self.vector_db.search(query, top_k)
    
    def generate_answer(self, query: str, top_k: int = 5) -> Dict:
        """Step 6: Generate answer with retrieved context"""
        # Retrieve documents
        documents = self.retrieve(query, top_k)
        
        if not documents:
            return {
                'answer': "No relevant information found.",
                'sources': []
            }
        
        # Build context
        context = "\n\n".join([
            f"[Source {i+1}]:\n{doc['text']}"
            for i, doc in enumerate(documents)
        ])
        
        # Create prompt
        prompt = f"""Answer the question based on the context below.
If the answer is not in the context, say "I don't know."

Context:
{context}

Question: {query}

Answer:"""
        
        # Generate answer
        response = self.llm_client.chat_completion([
            {"role": "system", "content": "You are a helpful assistant that answers based on provided context."},
            {"role": "user", "content": prompt}
        ], temperature=0.7)
        
        # Extract sources
        sources = list(set([
            doc['metadata']['source'] 
            for doc in documents
        ]))
        
        return {
            'answer': response['content'],
            'sources': sources,
            'retrieved_docs': documents
        }

# Usage
rag = RAGPipeline('knowledge-base')

# Ingest documents
documents = [
    {
        'id': 'policy_001',
        'content': 'Personal loan eligibility requires...',
        'source': 'loan_policy.pdf',

]

# Query

Advanced RAG: Reranking

from sentence_transformers import CrossEncoder

class AdvancedRAG(RAGPipeline):
    """RAG with two-stage retrieval and reranking"""
    
    def __init__(self, index_name: str):
        super().__init__(index_name)
        self.reranker = CrossEncoder('cross-encoder/ms-marco-MiniLM-L-6-v2')
    
    def retrieve_and_rerank(self, query: str, 
                           initial_k: int = 20,
                           final_k: int = 5) -> List[Dict]:
        """Two-stage retrieval with reranking"""
        # Stage 1: Vector search
        candidates = self.vector_db.search(query, top_k=initial_k)
        
        # Stage 2: Rerank with cross-encoder
        pairs = [[query, doc['text']] for doc in candidates]
        rerank_scores = self.reranker.predict(pairs)
        
        # Combine scores
        for i, doc in enumerate(candidates):
            doc['vector_score'] = doc['score']
            doc['rerank_score'] = float(rerank_scores[i])
            doc['combined_score'] = (
                0.5 * doc['vector_score'] + 
                0.5 * doc['rerank_score']
            )
        
        # Sort by combined score
        candidates.sort(key=lambda x: x['combined_score'], reverse=True)
        
        return candidates[:final_k]

Conversational RAG with Memory

class ConversationalRAG(RAGPipeline):
    """RAG with conversation history"""
    
    def __init__(self, index_name: str):
        super().__init__(index_name)
        self.conversation_history = []
        self.max_history = 5
    
    def chat(self, query: str, top_k: int = 5) -> Dict:
        """Chat with conversation context"""
        # Retrieve documents
        documents = self.retrieve(query, top_k)
        
        # Build context
        context = "\n\n".join([doc['text'] for doc in documents])
        
        # Build conversation history
        history_text = "\n".join([
            f"Human: {q}\nAssistant: {a}"
            for q, a in self.conversation_history[-self.max_history:]
        ])
        
        # Create prompt with history
        prompt = f"""Previous conversation:
{history_text}

Relevant context:
{context}

Current question: {query}

Answer based on the context and conversation history:"""
        
        # Generate response
        response = self.llm_client.chat_completion([
            {"role": "system", "content": "You are a helpful assistant with memory."},
            {"role": "user", "content": prompt}
        ])
        
        answer = response['content']
        
        # Update history
        self.conversation_history.append((query, answer))
        
        return {
            'answer': answer,
            'sources': [doc['metadata']['source'] for doc in documents]
        }
    
    def reset(self):
        """Clear conversation history"""
        self.conversation_history = []
💡 RAG Best Practices:
  • Chunk size: 500-1000 tokens with 100-200 overlap
  • Retrieve 5-10 documents initially
  • Use reranking for better relevance
  • Include metadata for filtering
  • Monitor and evaluate retrieval quality
  • Handle "I don't know" cases gracefully

5. Serverless Computing Basics

What is Serverless?

Execute code without managing servers. Cloud provider handles infrastructure, scaling, and maintenance. Pay only for actual execution time.

Key Benefits

  • No server management: Focus on code, not infrastructure
  • Auto-scaling: Handles 0 to millions of requests
  • Pay-per-use: Only pay for execution time
  • High availability: Built-in redundancy
  • Faster development: Less operational overhead

    AWS Lambda handler function

def api_handler(event, context):
    """Handle API Gateway requests"""
    
    # Extract request details
    http_method = event['httpMethod']
    path = event['path']
    query_params = event.get('queryStringParameters', {})
    headers = event.get('headers', {})
    body = json.loads(event.get('body', '{}'))
    
    # Route based on method and path
    if http_method == 'GET' and path == '/users':
        return get_users(query_params)
    
    elif http_method == 'POST' and path == '/users':
        return create_user(body)
    
    elif http_method == 'GET' and path.startswith('/users/'):
        user_id = path.split('/')[-1]
        return get_user(user_id)
    
    else:
        return {
            'statusCode': 404,
            'body': json.dumps({'error': 'Not found'})
        }

def s3_event_handler(event, context):
    """Process S3 object creation events"""
    
    for record in event['Records']:
        bucket = record['s3']['bucket']['name']
        key = record['s3']['object']['key']
        
        print(f"Processing: s3://{bucket}/{key}")
        
        # Download and process file
        obj = s3_client.get_object(Bucket=bucket, Key=key)
        content = obj['Body'].read()
        
        # Process content
        result = process_file(content)
        
        # Store result
        output_key = f"processed/{key}"
        s3_client.put_object(
            Bucket=bucket,
            Key=output_key,
            Body=json.dumps(result)
        )

# SQS Event Handler
def sqs_event_handler(event, context):
    """Process SQS messages in batches"""
    
    batch_item_failures = []
    
    for record in event['Records']:
        message_id = record['messageId']
        body = json.loads(record['body'])
        
        try:
            # Process message
            process_message(body)
            
        except Exception as e:
            print(f"Error processing {message_id}: {e}")
            # Add to failures for retry
            batch_item_failures.append({
                'itemIdentifier': message_id
            })
    
    return {
        'batchItemFailures': batch_item_failures
    }

# Cache expensive operations
@lru_cache(maxsize=128)
def get_config(key: str):
    """Cached configuration"""
    table = dynamodb.Table(os.environ['CONFIG_TABLE'])
    response = table.get_item(Key={'key': key})
    return response.get('Item', {}).get('value')

def lambda_handler(event, context):
    """Optimized handler"""
    start_time = time.time()
    
    # Check if cold start
    is_cold_start = not hasattr(lambda_handler, 'count')
    if is_cold_start:
        lambda_handler.count = 0
        print("COLD START")
    
    lambda_handler.count += 1
    
    # Use cached config
    api_key = get_config('api_key')
    
    # Use connection pool
    with db_engine.connect() as conn:
        result = conn.execute("SELECT COUNT(*) FROM users")
        count = result.scalar()
    
    execution_time = (time.time() - start_time) * 1000
    
    return {
        'statusCode': 200,
        'body': json.dumps({
            'cold_start': is_cold_start,
            'invocation': lambda_handler.count,
            'execution_ms': execution_time,
            'user_count': count
        })
    }

    """Singleton database connection"""

            cls._instance = super().__new__(cls)
            cls._engine = create_engine(
                os.environ['DATABASE_URL'],
                pool_pre_ping=True,
                pool_size=1
            )
            cls._session_factory = sessionmaker(bind=cls._engine)
        return cls._instance
    
    @contextmanager
    def get_session(self):
        """Context manager for sessions"""
        session = self._session_factory()
        try:
            yield session
            session.commit()
        except:
            session.rollback()
            raise
        finally:
            session.close()

        user_id = payload['user_id']
        role = payload.get('role', 'user')
        
        # Generate IAM policy
        policy = {
            'principalId': user_id,
            'policyDocument': {
                'Version': '2012-10-17',
                'Statement': [{
                    'Action': 'execute-api:Invoke',
                    'Effect': 'Allow',
                    'Resource': method_arn
                }]
            },
            'context': {
                'user_id': user_id,
                'role': role
            }
        }
        
        return policy
        
    except jwt.ExpiredSignatureError:
        raise Exception('Unauthorized: Token expired')
    except jwt.InvalidTokenError:
        raise Exception('Unauthorized: Invalid token')

from dataclasses import dataclass
from typing import Dict, List
import time

@dataclass
class APICall:
    timestamp: float
    model: str
    prompt_tokens: int
    completion_tokens: int
    total_tokens: int
    cost: float
    user_id: str = None

class LLMCostTracker:
    """Track and analyze LLM costs"""
    
    # Pricing per 1K tokens (2024)
    PRICING = {
        'gpt-4-turbo-preview': {'input': 0.01, 'output': 0.03},
        'gpt-4': {'input': 0.03, 'output': 0.06},
        'gpt-3.5-turbo': {'input': 0.0005, 'output': 0.0015},
        'claude-3-opus': {'input': 0.015, 'output': 0.075},
        'claude-3-sonnet': {'input': 0.003, 'output': 0.015},
        'claude-3-haiku': {'input': 0.00025, 'output': 0.00125},
        'text-embedding-3-small': {'input': 0.00002, 'output': 0},
    }
    
    def __init__(self):
        self.calls: List[APICall] = []
    
    def calculate_cost(self, model: str, 
                      prompt_tokens: int, 
                      completion_tokens: int) -> float:
        """Calculate cost for API call"""
        pricing = self.PRICING.get(model, {'input': 0, 'output': 0})
        
        input_cost = (prompt_tokens / 1000) * pricing['input']
        output_cost = (completion_tokens / 1000) * pricing['output']
        
        return input_cost + output_cost
    
    def track_call(self, model: str, usage: Dict, 
                  user_id: str = None):
        """Track API call"""
        cost = self.calculate_cost(
            model,
            usage['prompt_tokens'],
            usage.get('completion_tokens', 0)
        )
        
        call = APICall(
            timestamp=time.time(),
            model=model,
            prompt_tokens=usage['prompt_tokens'],
            completion_tokens=usage.get('completion_tokens', 0),
            total_tokens=usage['total_tokens'],
            cost=cost,
            user_id=user_id
        )
        
        self.calls.append(call)
    
    def get_stats(self) -> Dict:
        """Get usage statistics"""
        if not self.calls:
            return {}
        
        total_cost = sum(c.cost for c in self.calls)
        total_tokens = sum(c.total_tokens for c in self.calls)
        
        # Cost by model
        cost_by_model = {}
        for call in self.calls:
            model = call.model
            cost_by_model[model] = cost_by_model.get(model, 0) + call.cost
        
        # Cost by user
        cost_by_user = {}
        for call in self.calls:
            if call.user_id:
                user = call.user_id
                cost_by_user[user] = cost_by_user.get(user, 0) + call.cost
        
        return {
            'total_calls': len(self.calls),
            'total_cost': total_cost,
            'total_tokens': total_tokens,
            'avg_cost_per_call': total_cost / len(self.calls),
            'cost_by_model': cost_by_model,
            'cost_by_user': cost_by_user
        }

Caching Strategies

import hashlib
import json
from typing import Dict, Optional

class LLMCache:
    """Cache LLM responses to reduce costs"""
    
    def __init__(self):
        self.cache = {}  # In production: use Redis
    
    def _generate_key(self, messages: List[Dict], 
                     model: str, **params) -> str:
        """Generate cache key from request"""
        cache_input = {
            'messages': messages,
            'model': model,
            **params
        }
        
        # Create hash
        key_string = json.dumps(cache_input, sort_keys=True)
        return hashlib.sha256(key_string.encode()).hexdigest()
    
    def get(self, messages: List[Dict], model: str, 
           **params) -> Optional[Dict]:
        """Get cached response"""
        key = self._generate_key(messages, model, **params)
        return self.cache.get(key)
    
    def set(self, messages: List[Dict], model: str, 
           response: Dict, **params):
        """Cache response"""
        key = self._generate_key(messages, model, **params)
        self.cache[key] = response

class CachedLLMClient(OpenAIClient):
    """LLM client with caching"""
    
    def __init__(self, model: str = "gpt-4-turbo-preview"):
        super().__init__(model)
        self.cache = LLMCache()
    
    def chat_completion(self, messages: List[Dict], 
                       temperature: float = 0.7,
                       use_cache: bool = True,
                       **kwargs) -> Dict:
        """Chat completion with caching"""
        
        # Check cache
        if use_cache:
            cached = self.cache.get(
                messages, self.model, 
                temperature=temperature, **kwargs
            )
            
            if cached:
                print("Cache hit!")
                return cached
        
        # Call API
        response = super().chat_completion(
            messages, temperature, **kwargs
        )
        
        # Cache response (only if deterministic)
        if use_cache and temperature == 0:
            self.cache.set(
                messages, self.model, response,
                temperature=temperature, **kwargs
            )
        
        return response

class SecretManager:
    """Secure secret management with AWS Secrets Manager"""
    
    def __init__(self):
        self.client = boto3.client('secretsmanager')
        self._cache = {}
    
    def get_secret(self, secret_name: str) -> str:
        """Retrieve secret from AWS Secrets Manager"""
        
        # Check cache
        if secret_name in self._cache:
            return self._cache[secret_name]
        
        try:
            response = self.client.get_secret_value(
                SecretId=secret_name
            )
            
            secret = response['SecretString']
            self._cache[secret_name] = secret
            
            return secret
            
        except ClientError as e:
            print(f"Error retrieving secret: {e}")
            raise

Rate Limiting

import time
from collections import defaultdict
from typing import Optional

class RateLimiter:
    """Rate limiting for API calls"""
    
    def __init__(self, max_requests: int = 100, 
                window_seconds: int = 60):
        self.max_requests = max_requests
        self.window_seconds = window_seconds
        self.requests = defaultdict(list)
    
    def is_allowed(self, user_id: str) -> bool:
        """Check if request is allowed"""
        now = time.time()
        
        # Clean old requests
        cutoff = now - self.window_seconds
        self.requests[user_id] = [
            timestamp for timestamp in self.requests[user_id]
            if timestamp > cutoff
        ]
        
        # Check limit
        if len(self.requests[user_id]) >= self.max_requests:
            return False
        
        # Record request
        self.requests[user_id].append(now)
        return True
    
    def get_remaining(self, user_id: str) -> int:
        """Get remaining requests"""
        now = time.time()
        cutoff = now - self.window_seconds
        
        valid_requests = [
            t for t in self.requests[user_id]
            if t > cutoff
        ]
        
        return max(0, self.max_requests - len(valid_requests))

import openai

class ContentModerator:
    """Moderate user inputs and LLM outputs"""
    
    @staticmethod
    def check_input(text: str) -> Dict:
        """Check input for harmful content"""
        response = openai.Moderation.create(input=text)
        result = response['results'][0]
        
        return {
            'flagged': result['flagged'],
            'categories': result['categories'],
            'scores': result['category_scores']
        }
    
    @staticmethod
    def is_safe(text: str, threshold: float = 0.5) -> bool:
        """Check if text is safe"""
        result = ContentModerator.check_input(text)
        
        if result['flagged']:
            return False
        
        # Check individual category scores
        for category, score in result['scores'].items():
            if score > threshold:
                return False
        
        return True
