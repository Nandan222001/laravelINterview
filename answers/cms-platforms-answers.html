// Save post meta
add_action('save_post_product', 'myplugin_save_meta', 10, 3);
function myplugin_save_meta($post_id, $post, $update) {
    if (!isset($_POST['myplugin_nonce']) || 
        !wp_verify_nonce($_POST['myplugin_nonce'], 'myplugin_save')) {
        return;
    }
    
    if (defined('DOING_AUTOSAVE') && DOING_AUTOSAVE) {
        return;
    }
    
    if (!current_user_can('edit_post', $post_id)) {
        return;
    }
    
    if (isset($_POST['product_price'])) {
        update_post_meta(
            $post_id,
            '_product_price',
            sanitize_text_field($_POST['product_price'])
        );
    }
}

// AJAX handler
add_action('wp_ajax_myplugin_load_more', 'myplugin_ajax_load_more');
add_action('wp_ajax_nopriv_myplugin_load_more', 'myplugin_ajax_load_more');
function myplugin_ajax_load_more() {
    check_ajax_referer('myplugin-nonce', 'nonce');
    
    $paged = isset($_POST['page']) ? absint($_POST['page']) : 1;
    $query = new WP_Query(array(
        'post_type' => 'product',
        'posts_per_page' => 10,
        'paged' => $paged,
    ));
    
    if ($query->have_posts()) {
        ob_start();
        while ($query->have_posts()) {
            $query->the_post();
            get_template_part('template-parts/content', 'product');
        }
        $html = ob_get_clean();
        wp_reset_postdata();
        
        wp_send_json_success(array(
            'html' => $html,
            'has_more' => $paged < $query->max_num_pages,
        ));
    } else {
        wp_send_json_error();
    }
}

// Modify post title

5. WordPress REST API

Creating Custom REST API Endpoints

<?php
add_action('rest_api_init', 'myplugin_register_rest_routes');
function myplugin_register_rest_routes() {
    // GET endpoint
    register_rest_route('myplugin/v1', '/products', array(
        'methods' => 'GET',
        'callback' => 'myplugin_get_products',
        'permission_callback' => '__return_true',
        'args' => array(
            'per_page' => array(
                'default' => 10,
                'sanitize_callback' => 'absint',
            ),
        ),
    ));

    // POST endpoint
    register_rest_route('myplugin/v1', '/products', array(
        'methods' => 'POST',
        'callback' => 'myplugin_create_product',
        'permission_callback' => function() {
            return current_user_can('edit_posts');
        },
        'args' => array(
            'title' => array(
                'required' => true,
                'type' => 'string',
                'sanitize_callback' => 'sanitize_text_field',
            ),
            'price' => array(
                'required' => true,
                'type' => 'number',
            ),
        ),
    ));
}

function myplugin_get_products($request) {
    $per_page = $request->get_param('per_page');
    
    $query = new WP_Query(array(
        'post_type' => 'product',
        'posts_per_page' => $per_page,
        'post_status' => 'publish',
    ));
    
    $products = array();
    if ($query->have_posts()) {
        while ($query->have_posts()) {
            $query->the_post();
            $products[] = array(
                'id' => get_the_ID(),
                'title' => get_the_title(),
                'price' => get_post_meta(get_the_ID(), '_product_price', true),
                'permalink' => get_permalink(),
            );
        }
        wp_reset_postdata();
    }
    
    return new WP_REST_Response($products, 200);
}

function myplugin_create_product($request) {
    $title = $request->get_param('title');
    $price = $request->get_param('price');
    
    $product_id = wp_insert_post(array(
        'post_title' => $title,
        'post_type' => 'product',
        'post_status' => 'publish',
    ));
    
    if (is_wp_error($product_id)) {
        return new WP_Error('creation_failed', 'Failed to create product', array('status' => 500));
    }
    
    update_post_meta($product_id, '_product_price', $price);
    
    return new WP_REST_Response(array(
        'id' => $product_id,
        'message' => 'Product created successfully',
    ), 201);
}
💡 Authentication: For authenticated requests, use Application Passwords, JWT tokens, or OAuth. Always validate permissions.

What is EAV?

Entity-Attribute-Value (EAV) is a data model that allows dynamic attribute management without modifying database schema. Used for products, categories, and customers.

Creating Custom EAV Attribute

<?php
namespace Vendor\Module\Setup\Patch\Data;

use Magento\Eav\Setup\EavSetup;

{
    private $moduleDataSetup;
    private $eavSetupFactory;

        ModuleDataSetupInterface $moduleDataSetup,

        $this->moduleDataSetup = $moduleDataSetup;
        $this->eavSetupFactory = $eavSetupFactory;

                'group' => 'General',
            ]
        );
    }

    public static function getDependencies()
    {
        return [];
    }

    public function getAliases()
    {
        return [];
    }
}

10. Plugins vs Observers

Plugins (Interceptors)

Plugins allow you to modify the behavior of public methods by intercepting function calls.

Plugin Implementation

<?php
namespace Vendor\Module\Plugin;

use Magento\Catalog\Model\Product;

class ProductPlugin
{
    public function beforeSetName(Product $subject, $name)
    {
        $name = strtoupper($name);
        return [$name];
    }

    public function afterGetName(Product $subject, $result)
    {
        $result .= ' - Modified';
        return $result;
    }

    public function aroundSave(Product $subject, callable $proceed)
    {
        // Code before
        $this->doSomethingBefore();
        
        // Call original method
        $result = $proceed();
        
        // Code after
        $this->doSomethingAfter();
        
        return $result;
    }

    private function doSomethingBefore() {}
    private function doSomethingAfter() {}
}

11. Payment Integration

Payment Method Model

<?php
namespace Vendor\Module\Model;

use Magento\Payment\Model\Method\AbstractMethod;
use Magento\Payment\Model\InfoInterface;

class Payment extends AbstractMethod
{
    const CODE = 'custompayment';
    
    protected $_code = self::CODE;
    protected $_canAuthorize = true;
    protected $_canCapture = true;
    protected $_canRefund = true;

    public function authorize(InfoInterface $payment, $amount)
    {
        $transactionId = $this->processAuthorization($payment, $amount);
        $payment->setTransactionId($transactionId);
        $payment->setIsTransactionClosed(false);
        return $this;
    }

    public function capture(InfoInterface $payment, $amount)
    {
        $transactionId = $this->processCapture($payment, $amount);
        $payment->setTransactionId($transactionId);
        $payment->setIsTransactionClosed(true);
        return $this;
    }

    public function refund(InfoInterface $payment, $amount)
    {
        $transactionId = $this->processRefund($payment, $amount);
        $payment->setTransactionId($transactionId . '-refund');
        return $this;
    }

    private function processAuthorization($payment, $amount)
    {
        // Call payment gateway API
        return 'AUTH-' . time();
    }

    private function processCapture($payment, $amount)
    {
        return 'CAPTURE-' . time();
    }

    private function processRefund($payment, $amount)
    {
        return 'REFUND-' . time();
    }
}
💡 Security: Never store sensitive payment data. Use tokenization and comply with PCI-DSS standards.

14. Drupal Form API

Creating a Form

<?php
namespace Drupal\custom_module\Form;

use Drupal\Core\Form\FormBase;
use Drupal\Core\Form\FormStateInterface;

class CustomForm extends FormBase {

    public function getFormId() {
        return 'custom_module_custom_form';
    }

    public function buildForm(array $form, FormStateInterface $form_state) {
        $form['name'] = [
            '#type' => 'textfield',
            '#title' => $this->t('Name'),
            '#required' => TRUE,
            '#maxlength' => 255,
        ];

        $form['email'] = [
            '#type' => 'email',
            '#title' => $this->t('Email'),
            '#required' => TRUE,
        ];

        $form['message'] = [
            '#type' => 'textarea',
            '#title' => $this->t('Message'),
            '#required' => TRUE,
            '#rows' => 5,
        ];

        $form['submit'] = [
            '#type' => 'submit',
            '#value' => $this->t('Submit'),
        ];

        return $form;
    }

    public function validateForm(array &$form, FormStateInterface $form_state) {
        $name = $form_state->getValue('name');
        if (strlen($name) < 3) {
            $form_state->setErrorByName('name', 
                $this->t('Name must be at least 3 characters.')
            );
        }
    }

    public function submitForm(array &$form, FormStateInterface $form_state) {
        $values = $form_state->getValues();
        
        $this->messenger()->addMessage(
            $this->t('Thank you @name!', ['@name' => $values['name']])
        );
    }
}

Configuration Form

<?php
namespace Drupal\custom_module\Form;

use Drupal\Core\Form\ConfigFormBase;
use Drupal\Core\Form\FormStateInterface;

class SettingsForm extends ConfigFormBase {

    public function getFormId() {
        return 'custom_module_settings_form';
    }

    protected function getEditableConfigNames() {
        return ['custom_module.settings'];
    }

    public function buildForm(array $form, FormStateInterface $form_state) {
        $config = $this->config('custom_module.settings');

        $form['api_key'] = [
            '#type' => 'textfield',
            '#title' => $this->t('API Key'),
            '#default_value' => $config->get('api_key'),
        ];

        return parent::buildForm($form, $form_state);
    }

    public function submitForm(array &$form, FormStateInterface $form_state) {
        $this->config('custom_module.settings')
            ->set('api_key', $form_state->getValue('api_key'))
            ->save();

        parent::submitForm($form, $form_state);
    }
}

Custom Twig Extension

<?php
namespace Drupal\custom_module\TwigExtension;

use Twig\Extension\AbstractExtension;
use Twig\TwigFilter;
use Twig\TwigFunction;

class CustomTwigExtension extends AbstractExtension {

    public function getFilters() {
        return [
            new TwigFilter('custom_filter', [$this, 'customFilter']),
        ];
    }

    public function getFunctions() {
        return [
            new TwigFunction('custom_function', [$this, 'customFunction']),
        ];
    }

    public function customFilter($string) {
        return strtoupper($string);
    }

    public function customFunction($arg) {
        return 'Custom: ' . $arg;
    }
}
