1. Magento 2 Architecture

Overview

Magento 2 follows a modular architecture with service contracts, dependency injection, and a layered design pattern.

Architecture Layers

Layer Purpose Components
Presentation Layer User interface and web APIs Layouts, blocks, templates, controllers
Service Layer Business logic and service contracts Service contracts, APIs
Domain Layer Business logic implementation Models, resource models, collections
Persistence Layer Data access and storage Database operations, indexers

Module Structure

Vendor/Module/
├── Api/                    # Service contracts
│   ├── Data/              # Data interfaces
│   └── RepositoryInterface.php
├── Block/                 # View blocks
├── Console/               # CLI commands
├── Controller/            # Controllers
│   ├── Adminhtml/        # Admin controllers
│   └── Index/            # Frontend controllers
├── Cron/                 # Cron jobs
├── etc/                  # Configuration
│   ├── module.xml        # Module declaration
│   ├── di.xml            # Dependency injection
│   ├── routes.xml        # URL routes
│   ├── acl.xml           # Access control
│   └── config.xml        # Default config
├── Helper/               # Helper classes
├── Model/                # Business logic
│   ├── ResourceModel/    # Database operations
│   └── Repository.php    # Repository implementation
├── Observer/             # Event observers

2. EAV (Entity-Attribute-Value) Model

What is EAV?

EAV is a data model that allows dynamic attribute management without modifying database schema. Used extensively for products, categories, and customers.

EAV Database Structure

Table Type Purpose Example
Entity Main entity data catalog_product_entity
Attribute Attribute metadata eav_attribute
Value Tables Attribute values by type catalog_product_entity_int, _varchar, _text, _decimal, _datetime

Creating Custom EAV Attribute

<?php
namespace Vendor\Module\Setup\Patch\Data;

use Magento\Eav\Setup\EavSetup;
use Magento\Eav\Setup\EavSetupFactory;
use Magento\Framework\Setup\ModuleDataSetupInterface;
use Magento\Framework\Setup\Patch\DataPatchInterface;

class AddCustomAttribute implements DataPatchInterface
{
    private $moduleDataSetup;
    private $eavSetupFactory;

    public function __construct(
        ModuleDataSetupInterface $moduleDataSetup,
        EavSetupFactory $eavSetupFactory
    ) {
        $this->moduleDataSetup = $moduleDataSetup;
        $this->eavSetupFactory = $eavSetupFactory;
    }

    public function apply()
    {
        $eavSetup = $this->eavSetupFactory->create(['setup' => $this->moduleDataSetup]);
        
        $eavSetup->addAttribute(
            \Magento\Catalog\Model\Product::ENTITY,
            'custom_attribute',
            [
                'type' => 'varchar',
                'label' => 'Custom Attribute',
                'input' => 'text',
                'required' => false,
                'sort_order' => 100,
                'global' => \Magento\Eav\Model\Entity\Attribute\ScopedAttributeInterface::SCOPE_GLOBAL,
                'used_in_product_listing' => true,
                'visible_on_front' => true,
                'user_defined' => true,
                'apply_to' => 'simple,configurable',
                'group' => 'General',
            ]
        );
    }

    public static function getDependencies()
    {
        return [];
    }

    public function getAliases()
    {
        return [];
    }
}
⚠️ Performance: EAV queries can be slow due to multiple table joins. Use flat tables for product/category data in production, and consider using collections with proper attribute filtering.

3. Plugins vs Observers

Plugins (Interceptors)

Plugins allow you to modify the behavior of public methods by intercepting function calls.

Plugin Types

  • Before: Runs before the observed method
  • After: Runs after the observed method, can modify result
  • Around: Wraps around the observed method, full control

Plugin Implementation

<?php
namespace Vendor\Module\Plugin;

use Magento\Catalog\Model\Product;

class ProductPlugin
{
    public function beforeSetName(Product $subject, $name)
    {
        $name = strtoupper($name);
        return [$name];
    }

    public function afterGetName(Product $subject, $result)
    {
        $result .= ' - Modified';
        return $result;
    }

    public function aroundSave(Product $subject, callable $proceed)
    {
        $this->doSomethingBefore();
        $result = $proceed();
        $this->doSomethingAfter();
        return $result;
    }

    private function doSomethingBefore() {}
    private function doSomethingAfter() {}
}

Observers (Events)

<?php
namespace Vendor\Module\Observer;

use Magento\Framework\Event\Observer;
use Magento\Framework\Event\ObserverInterface;

class ProductSaveAfter implements ObserverInterface
{
    public function execute(Observer $observer)
    {
        $product = $observer->getEvent()->getProduct();
        // Custom logic here
    }
}

Comparison

Feature Plugins Observers
Use Case Modify method behavior React to events
Can modify result Yes (after/around) No
Works on Public methods only Dispatched events

4. Payment Gateway Integration

Payment Method Model

<?php
namespace Vendor\Module\Model;

use Magento\Payment\Model\Method\AbstractMethod;

class Payment extends AbstractMethod
{
    const CODE = 'custompayment';
    
    protected $_code = self::CODE;
    protected $_canAuthorize = true;
    protected $_canCapture = true;
    protected $_canRefund = true;

    public function authorize(InfoInterface $payment, $amount)
    {
        $transactionId = $this->processAuthorization($payment, $amount);
        $payment->setTransactionId($transactionId);
        $payment->setIsTransactionClosed(false);
        return $this;
    }

    public function capture(InfoInterface $payment, $amount)
    {
        $transactionId = $this->processCapture($payment, $amount);
        $payment->setTransactionId($transactionId);
        $payment->setIsTransactionClosed(true);
        return $this;
    }

    public function refund(InfoInterface $payment, $amount)
    {
        $transactionId = $this->processRefund($payment, $amount);
        $payment->setTransactionId($transactionId . '-refund');
        return $this;
    }

    private function processAuthorization($payment, $amount)
    {
        return 'AUTH-' . time();
    }

    private function processCapture($payment, $amount)
    {
        return 'CAPTURE-' . time();
    }

    private function processRefund($payment, $amount)
    {
        return 'REFUND-' . time();
    }
}
💡 Security: Never store sensitive payment data (card numbers, CVV) in Magento. Use tokenization and comply with PCI-DSS standards.

5. GraphQL

GraphQL Schema Definition

type Query {
    customProducts(
        filter: ProductFilterInput
        pageSize: Int = 20
        currentPage: Int = 1
    ): CustomProductsOutput @resolver(class: "Vendor\\Module\\Model\\Resolver\\Products")
}

type CustomProduct {
    id: Int
    name: String
    sku: String
    price: Float
    description: String
}

Query Resolver

<?php
namespace Vendor\Module\Model\Resolver;

use Magento\Framework\GraphQl\Query\ResolverInterface;

class Products implements ResolverInterface
{
    private $productRepository;

    public function __construct(
        ProductRepositoryInterface $productRepository
    ) {
        $this->productRepository = $productRepository;
    }

    public function resolve($field, $context, $info, $value = null, $args = null)
    {
        $searchResult = $this->productRepository->getList($searchCriteria);
        
        $items = [];
        foreach ($searchResult->getItems() as $product) {
            $items[] = [
                'id' => $product->getId(),
                'name' => $product->getName(),
                'sku' => $product->getSku(),
                'price' => $product->getPrice(),
            ];
        }
        
        return ['items' => $items];
    }
}
✅ Performance: GraphQL allows clients to request only the data they need, reducing payload size and improving performance.

7. Drupal Hooks System

Common Hooks

<?php

function custom_module_form_alter(&$form, FormStateInterface $form_state, $form_id) {
    if ($form_id == 'node_article_form') {
        $form['title']['widget'][0]['value']['#description'] = t('Custom description');
    }
}

function custom_module_entity_presave(EntityInterface $entity) {
    if ($entity->getEntityTypeId() == 'node') {
        $entity->set('field_custom', 'Custom value');
    }
}

function custom_module_entity_insert(EntityInterface $entity) {
    if ($entity instanceof NodeInterface) {
        \Drupal::logger('custom_module')->notice('Node created: @title', [
            '@title' => $entity->getTitle()
        ]);
    }
}

function custom_module_preprocess_node(&$variables) {
    $node = $variables['node'];
    $variables['custom_variable'] = 'Custom value';
}

function custom_module_cron() {
    $last_run = \Drupal::state()->get('custom_module.last_cron', 0);
    $now = \Drupal::time()->getRequestTime();
    
    if ($now - $last_run > 3600) {
        \Drupal::state()->set('custom_module.last_cron', $now);
    }
}

Hook Types

Hook When It Runs Common Use
hook_form_alter() Before form renders Modify form structure
hook_entity_presave() Before entity saves Modify entity data
hook_entity_insert() After entity created React to new entities
hook_preprocess_HOOK() Before template renders Add template variables

8. Form API

Creating a Form

<?php
namespace Drupal\custom_module\Form;

use Drupal\Core\Form\FormBase;
use Drupal\Core\Form\FormStateInterface;

class CustomForm extends FormBase {

    public function getFormId() {
        return 'custom_module_custom_form';
    }

    public function buildForm(array $form, FormStateInterface $form_state) {
        $form['name'] = [
            '#type' => 'textfield',
            '#title' => $this->t('Name'),
            '#required' => TRUE,
            '#maxlength' => 255,
        ];

        $form['email'] = [
            '#type' => 'email',
            '#title' => $this->t('Email'),
            '#required' => TRUE,
        ];

        $form['message'] = [
            '#type' => 'textarea',
            '#title' => $this->t('Message'),
            '#required' => TRUE,
            '#rows' => 5,
        ];

        $form['submit'] = [
            '#type' => 'submit',
            '#value' => $this->t('Submit'),
        ];

        return $form;
    }

    public function validateForm(array &$form, FormStateInterface $form_state) {
        $name = $form_state->getValue('name');
        if (strlen($name) < 3) {
            $form_state->setErrorByName('name', 
                $this->t('Name must be at least 3 characters.')
            );
        }
    }

    public function submitForm(array &$form, FormStateInterface $form_state) {
        $values = $form_state->getValues();
        
        $this->messenger()->addMessage(
            $this->t('Thank you @name!', ['@name' => $values['name']])
        );
    }
}

Configuration Form

<?php
namespace Drupal\custom_module\Form;

use Drupal\Core\Form\ConfigFormBase;
use Drupal\Core\Form\FormStateInterface;

class SettingsForm extends ConfigFormBase {

    public function getFormId() {
        return 'custom_module_settings_form';
    }

    protected function getEditableConfigNames() {
        return ['custom_module.settings'];
    }

    public function buildForm(array $form, FormStateInterface $form_state) {
        $config = $this->config('custom_module.settings');

        $form['api_key'] = [
            '#type' => 'textfield',
            '#title' => $this->t('API Key'),
            '#default_value' => $config->get('api_key'),
        ];

        return parent::buildForm($form, $form_state);
    }

    public function submitForm(array &$form, FormStateInterface $form_state) {
        $this->config('custom_module.settings')
            ->set('api_key', $form_state->getValue('api_key'))
            ->save();

        parent::submitForm($form, $form_state);
    }
}

What are Views?

Views is Drupal's powerful query builder that allows you to create, manage, and display lists of content without writing code.

Creating Views Programmatically

Custom Twig Extension

<?php
namespace Drupal\custom_module\TwigExtension;

use Twig\Extension\AbstractExtension;
use Twig\TwigFilter;
use Twig\TwigFunction;

class CustomTwigExtension extends AbstractExtension {

    public function getFilters() {
        return [
            new TwigFilter('custom_filter', [$this, 'customFilter']),
        ];
    }

    public function getFunctions() {
        return [
            new TwigFunction('custom_function', [$this, 'customFunction']),
        ];
    }

    public function customFilter($string) {
        return strtoupper($string);
    }

    public function customFunction($arg) {
        return 'Custom: ' . $arg;
    }
}

Register Twig Extension

# custom_module.services.yml
services:
  custom_module.twig_extension:
    class: Drupal\custom_module\TwigExtension\CustomTwigExtension
    tags:
      - { name: twig.extension }
✅ Security: Twig automatically escapes output to prevent XSS attacks. Use |raw filter only when absolutely necessary and the content is trusted.
