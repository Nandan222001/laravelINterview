1. Encapsulation

What is Encapsulation?

Encapsulation is the bundling of data (properties) and methods that operate on that data within a single unit (class), while restricting direct access to some components. It's about hiding internal implementation details and exposing only what's necessary through a public interface.

Key Benefits

  • Data Protection: Prevents unauthorized or unintended access to internal state
  • Flexibility: Internal implementation can change without affecting external code
  • Maintainability: Reduces coupling between components
  • Validation: Control how data is accessed and modified

Implementation Example

<?php
class BankAccount {
    private float $balance;
    private string $accountNumber;
    private array $transactionHistory = [];
    
    public function __construct(string $accountNumber, float $initialBalance = 0) {
        $this->accountNumber = $accountNumber;
        $this->balance = $initialBalance;
        $this->logTransaction('Account opened', $initialBalance);
    }
    
    public function deposit(float $amount): void {
        if ($amount <= 0) {
            throw new InvalidArgumentException("Deposit amount must be positive");
        }
        
        $this->balance += $amount;
        $this->logTransaction('Deposit', $amount);
    }
    
    public function withdraw(float $amount): void {
        if ($amount <= 0) {
            throw new InvalidArgumentException("Withdrawal amount must be positive");
        }
        
        if ($amount > $this->balance) {
            throw new RuntimeException("Insufficient funds");
        }
        
        $this->balance -= $amount;
        $this->logTransaction('Withdrawal', -$amount);
    }
    
    public function getBalance(): float {
        return $this->balance;
    }
    
    public function getAccountNumber(): string {
        return $this->accountNumber;
    }
    
    public function getTransactionHistory(): array {
        return $this->transactionHistory;
    }
    
    private function logTransaction(string $type, float $amount): void {
        $this->transactionHistory[] = [
            'type' => $type,
            'amount' => $amount,
            'balance' => $this->balance,
            'timestamp' => new DateTime()
        ];
    }
}

$account = new BankAccount('ACC-12345', 1000);
$account->deposit(500);
$account->withdraw(200);
echo "Balance: $" . $account->getBalance();
?>
💡 Best Practice: Always use private or protected for properties and expose public methods (getters/setters) only when necessary. This gives you control over data access and validation.

2. Abstraction

What is Abstraction?

Abstraction is the process of hiding complex implementation details and showing only essential features of an object. It focuses on what an object does rather than how it does it.

Abstract Classes

<?php
abstract class PaymentGateway {
    protected string $apiKey;
    protected bool $testMode;
    
    public function __construct(string $apiKey, bool $testMode = false) {
        $this->apiKey = $apiKey;
        $this->testMode = $testMode;
    }
    
    abstract public function processPayment(float $amount, array $data): array;
    abstract public function refund(string $transactionId, float $amount): array;
    abstract public function validateCredentials(): bool;
    
    public function log(string $message): void {
        $mode = $this->testMode ? 'TEST' : 'LIVE';
        error_log("[$mode] " . static::class . ": $message");
    }
    
    protected function generateTransactionId(): string {
        return uniqid('txn_', true);
    }
}

class StripeGateway extends PaymentGateway {
    public function processPayment(float $amount, array $data): array {
        $this->log("Processing payment of $$amount via Stripe");
        
        return [
            'success' => true,
            'transaction_id' => $this->generateTransactionId(),
            'amount' => $amount,
            'gateway' => 'stripe'
        ];
    }
    
    public function refund(string $transactionId, float $amount): array {
        $this->log("Refunding $$amount for transaction $transactionId");
        
        return [
            'success' => true,
            'refund_id' => $this->generateTransactionId(),
            'amount' => $amount
        ];
    }
    
    public function validateCredentials(): bool {
        return strlen($this->apiKey) > 0 && str_starts_with($this->apiKey, 'sk_');
    }
}

function makePayment(PaymentGateway $gateway, float $amount, array $data): void {
    if (!$gateway->validateCredentials()) {
        throw new Exception("Invalid gateway credentials");
    }
    
    $result = $gateway->processPayment($amount, $data);
    
    if ($result['success']) {
        echo "Payment successful: {$result['transaction_id']}\n";
    }
}

$stripe = new StripeGateway('sk_test_123', true);
makePayment($stripe, 100.00, ['card' => '4242424242424242']);
?>
✅ Key Takeaway: Abstraction allows you to define a contract (what must be done) without specifying the implementation (how it's done). This makes your code flexible and easier to extend.

3. Inheritance

What is Inheritance?

Inheritance is a mechanism where a new class (child/derived) inherits properties and methods from an existing class (parent/base). It promotes code reuse and establishes a relationship between classes.

Basic Inheritance

<?php
class Employee {
    protected string $name;
    protected string $email;
    protected float $salary;
    
    public function __construct(string $name, string $email, float $salary) {
        $this->name = $name;
        $this->email = $email;
        $this->salary = $salary;
    }
    
    public function getName(): string {
        return $this->name;
    }
    
    public function getSalary(): float {
        return $this->salary;
    }
    
    public function calculateBonus(): float {
        return $this->salary * 0.05;
    }
}

class Manager extends Employee {
    private array $team = [];
    private string $department;
    
    public function __construct(string $name, string $email, float $salary, string $department) {
        parent::__construct($name, $email, $salary);
        $this->department = $department;
    }
    
    public function addTeamMember(Employee $employee): void {
        $this->team[] = $employee;
    }
    
    public function getTeamSize(): int {
        return count($this->team);
    }
    
    public function calculateBonus(): float {
        $baseBonus = parent::calculateBonus();
        $teamBonus = $this->getTeamSize() * 500;
        return $baseBonus + $teamBonus;
    }
}

$employee = new Employee('John Doe', 'john@example.com', 50000);
echo "Employee bonus: $" . $employee->calculateBonus() . "\n";

$manager = new Manager('Jane Smith', 'jane@example.com', 80000, 'Engineering');
$manager->addTeamMember($employee);
echo "Manager bonus: $" . $manager->calculateBonus() . "\n";
?>
⚠️ Caution: Favor composition over inheritance when possible. Deep inheritance hierarchies can become difficult to maintain. Use inheritance when there's a true "is-a" relationship.

4. Polymorphism

What is Polymorphism?

Polymorphism means "many forms". It allows objects of different classes to be treated as objects of a common parent class or interface. The same method call can produce different behaviors depending on the object type.

Interface-Based Polymorphism

<?php
interface Shape {
    public function calculateArea(): float;
    public function calculatePerimeter(): float;
    public function getName(): string;
}

class Circle implements Shape {
    private float $radius;
    
    public function __construct(float $radius) {
        $this->radius = $radius;
    }
    
    public function calculateArea(): float {
        return pi() * $this->radius ** 2;
    }
    
    public function calculatePerimeter(): float {
        return 2 * pi() * $this->radius;
    }
    
    public function getName(): string {
        return "Circle";
    }
}

class Rectangle implements Shape {
    private float $width;
    private float $height;
    
    public function __construct(float $width, float $height) {
        $this->width = $width;
        $this->height = $height;
    }
    
    public function calculateArea(): float {
        return $this->width * $this->height;
    }
    
    public function calculatePerimeter(): float {
        return 2 * ($this->width + $this->height);
    }
    
    public function getName(): string {
        return "Rectangle";
    }
}

function printShapeInfo(Shape $shape): void {
    echo "Shape: {$shape->getName()}\n";
    echo "Area: " . number_format($shape->calculateArea(), 2) . "\n";
    echo "Perimeter: " . number_format($shape->calculatePerimeter(), 2) . "\n";
}

function calculateTotalArea(array $shapes): float {
    $total = 0;
    foreach ($shapes as $shape) {
        if ($shape instanceof Shape) {
            $total += $shape->calculateArea();
        }
    }
    return $total;
}

$shapes = [
    new Circle(5),
    new Rectangle(10, 5)
];

foreach ($shapes as $shape) {
    printShapeInfo($shape);
}

echo "Total area: " . calculateTotalArea($shapes) . "\n";
?>
✅ Key Benefit: Polymorphism allows you to write code that works with interfaces rather than concrete implementations, making your code more flexible and extensible.

5. SOLID: Single Responsibility Principle (SRP)

What is SRP?

A class should have only one reason to change, meaning it should have only one job or responsibility. Each class should focus on doing one thing well.

Good Example: Single Responsibility

<?php
class User {
    private string $name;
    private string $email;
    
    public function __construct(string $name, string $email) {
        $this->name = $name;
        $this->email = $email;
    }
    
    public function getName(): string {
        return $this->name;
    }
    
    public function getEmail(): string {
        return $this->email;
    }
}

class UserRepository {
    private PDO $connection;
    
    public function __construct(PDO $connection) {
        $this->connection = $connection;
    }
    
    public function save(User $user): void {
        $stmt = $this->connection->prepare(
            'INSERT INTO users (name, email) VALUES (?, ?)'
        );
        $stmt->execute([$user->getName(), $user->getEmail()]);
    }
    
    public function findByEmail(string $email): ?User {
        $stmt = $this->connection->prepare('SELECT * FROM users WHERE email = ?');
        $stmt->execute([$email]);
        $data = $stmt->fetch();
        
        return $data ? new User($data['name'], $data['email']) : null;
    }
}

class UserValidator {
    public function validate(User $user): bool {
        return $this->isValidEmail($user->getEmail()) && 
               $this->isValidName($user->getName());
    }
    
    private function isValidEmail(string $email): bool {
        return filter_var($email, FILTER_VALIDATE_EMAIL) !== false;
    }
    
    private function isValidName(string $name): bool {
        return !empty($name) && strlen($name) >= 2;
    }
}

class EmailService {
    private string $smtpHost;
    
    public function __construct(string $smtpHost) {
        $this->smtpHost = $smtpHost;
    }
    
    public function sendWelcomeEmail(User $user): void {
        $to = $user->getEmail();
        $subject = 'Welcome!';
        $message = "Hello {$user->getName()}, welcome to our application!";
        
        mail($to, $subject, $message);
    }
}

$pdo = new PDO('mysql:host=localhost;dbname=app', 'root', 'password');
$user = new User('John Doe', 'john@example.com');

$validator = new UserValidator();
if ($validator->validate($user)) {
    $repository = new UserRepository($pdo);
    $repository->save($user);
    
    $emailService = new EmailService('smtp.example.com');
    $emailService->sendWelcomeEmail($user);
}
?>
💡 Benefits of SRP:
  • Easier to understand and maintain
  • Easier to test (each class has focused tests)
  • Reduces coupling between components
  • Changes to one responsibility don't affect others

6. SOLID: Open/Closed Principle (OCP)

What is OCP?

Software entities should be open for extension but closed for modification. You should be able to add new functionality without changing existing code.

Good Example: Follows OCP

<?php
interface DiscountStrategy {
    public function calculate(float $amount): float;
    public function getName(): string;
}

class PercentageDiscount implements DiscountStrategy {
    private float $percentage;
    
    public function __construct(float $percentage) {
        $this->percentage = $percentage;
    }
    
    public function calculate(float $amount): float {
        return $amount * ($this->percentage / 100);
    }
    
    public function getName(): string {
        return "Percentage Discount ({$this->percentage}%)";
    }
}

class FixedDiscount implements DiscountStrategy {
    private float $discount;
    
    public function __construct(float $discount) {
        $this->discount = $discount;
    }
    
    public function calculate(float $amount): float {
        return min($this->discount, $amount);
    }
    
    public function getName(): string {
        return "Fixed Discount ($" . $this->discount . ")";
    }
}

class LoyaltyDiscount implements DiscountStrategy {
    private int $yearsAsCustomer;
    
    public function __construct(int $yearsAsCustomer) {
        $this->yearsAsCustomer = $yearsAsCustomer;
    }
    
    public function calculate(float $amount): float {
        $discountPercent = min($this->yearsAsCustomer * 2, 20);
        return $amount * ($discountPercent / 100);
    }
    
    public function getName(): string {
        return "Loyalty Discount";
    }
}

class Order {
    private float $amount;
    private ?DiscountStrategy $discount = null;
    
    public function __construct(float $amount) {
        $this->amount = $amount;
    }
    
    public function applyDiscount(DiscountStrategy $discount): void {
        $this->discount = $discount;
    }
    
    public function getTotal(): float {
        $discount = $this->discount ? $this->discount->calculate($this->amount) : 0;
        return max(0, $this->amount - $discount);
    }
}

$order = new Order(100);
$order->applyDiscount(new PercentageDiscount(15));
echo "Total: $" . $order->getTotal() . "\n";

$order->applyDiscount(new LoyaltyDiscount(5));
echo "Total: $" . $order->getTotal() . "\n";
?>
✅ Key Benefit: By using interfaces and polymorphism, you can add new functionality by creating new classes instead of modifying existing ones, reducing the risk of introducing bugs.

7. SOLID: Liskov Substitution Principle (LSP)

What is LSP?

Objects of a superclass should be replaceable with objects of a subclass without breaking the application. Subclasses should extend parent behavior, not replace or remove it.

Good Example: Follows LSP

<?php
interface Shape {
    public function getArea(): float;
    public function getPerimeter(): float;
}

class Rectangle implements Shape {
    private float $width;
    private float $height;
    
    public function __construct(float $width, float $height) {
        $this->width = $width;
        $this->height = $height;
    }
    
    public function getArea(): float {
        return $this->width * $this->height;
    }
    
    public function getPerimeter(): float {
        return 2 * ($this->width + $this->height);
    }
}

class Square implements Shape {
    private float $side;
    
    public function __construct(float $side) {
        $this->side = $side;
    }
    
    public function getArea(): float {
        return $this->side ** 2;
    }
    
    public function getPerimeter(): float {
        return 4 * $this->side;
    }
}

function calculateTotalArea(array $shapes): float {
    $total = 0;
    foreach ($shapes as $shape) {
        if ($shape instanceof Shape) {
            $total += $shape->getArea();
        }
    }
    return $total;
}

$shapes = [
    new Rectangle(5, 10),
    new Square(5)
];

echo "Total area: " . calculateTotalArea($shapes) . "\n";
?>
⚠️ Remember: If a subclass can't properly implement all methods of its parent, the inheritance relationship is likely wrong. Use interfaces to separate capabilities.

8. SOLID: Interface Segregation Principle (ISP)

What is ISP?

Clients should not be forced to depend on interfaces they don't use. It's better to have many specific interfaces than one general-purpose interface.

Good Example: Segregated Interfaces

<?php
interface Workable {
    public function work(): void;
}

interface Eatable {
    public function eat(): void;
}

interface Sleepable {
    public function sleep(): void;
}

interface Payable {
    public function getSalary(): float;
    public function receivePayment(float $amount): void;
}

class HumanWorker implements Workable, Eatable, Sleepable, Payable {
    private float $salary;
    
    public function __construct(float $salary) {
        $this->salary = $salary;
    }
    
    public function work(): void {
        echo "Human working...\n";
    }
    
    public function eat(): void {
        echo "Human eating...\n";
    }
    
    public function sleep(): void {
        echo "Human sleeping...\n";
    }
    
    public function getSalary(): float {
        return $this->salary;
    }
    
    public function receivePayment(float $amount): void {
        echo "Received payment: $$amount\n";
    }
}

class RobotWorker implements Workable {
    public function work(): void {
        echo "Robot working 24/7...\n";
    }
}

class WorkManager {
    public function manageWork(Workable $worker): void {
        $worker->work();
    }
    
    public function processPayroll(Payable $worker): void {
        $salary = $worker->getSalary();
        $worker->receivePayment($salary);
    }
    
    public function scheduleBreak(Eatable $worker): void {
        $worker->eat();
    }
}

$human = new HumanWorker(5000);
$robot = new RobotWorker();

$manager = new WorkManager();
$manager->manageWork($human);
$manager->manageWork($robot);
$manager->processPayroll($human);
$manager->scheduleBreak($human);
?>

9. SOLID: Dependency Inversion Principle (DIP)

What is DIP?

High-level modules should not depend on low-level modules. Both should depend on abstractions (interfaces).

Good Example: Depend on Abstractions

<?php
interface DatabaseInterface {
    public function connect(): void;
    public function query(string $sql): array;
    public function execute(string $sql): bool;
}

class MySQLDatabase implements DatabaseInterface {
    private string $host;
    
    public function __construct(string $host) {
        $this->host = $host;
    }
    
    public function connect(): void {
        echo "Connecting to MySQL at {$this->host}...\n";
    }
    
    public function query(string $sql): array {
        echo "Executing MySQL query: $sql\n";
        return [];
    }
    
    public function execute(string $sql): bool {
        echo "Executing MySQL statement: $sql\n";
        return true;
    }
}

class PostgreSQLDatabase implements DatabaseInterface {
    private string $connectionString;
    
    public function __construct(string $connectionString) {
        $this->connectionString = $connectionString;
    }
    
    public function connect(): void {
        echo "Connecting to PostgreSQL...\n";
    }
    
    public function query(string $sql): array {
        echo "Executing PostgreSQL query: $sql\n";
        return [];
    }
    
    public function execute(string $sql): bool {
        echo "Executing PostgreSQL statement: $sql\n";
        return true;
    }
}

class UserRepository {
    private DatabaseInterface $database;
    
    public function __construct(DatabaseInterface $database) {
        $this->database = $database;
    }
    
    public function findAll(): array {
        return $this->database->query('SELECT * FROM users');
    }
    
    public function create(array $data): bool {
        $sql = "INSERT INTO users (name, email) VALUES ('{$data['name']}', '{$data['email']}')";
        return $this->database->execute($sql);
    }
}

$mysqlDb = new MySQLDatabase('localhost');
$postgresDb = new PostgreSQLDatabase('postgresql://localhost/myapp');

$userRepo1 = new UserRepository($mysqlDb);
$userRepo2 = new UserRepository($postgresDb);

$userRepo1->findAll();
$userRepo2->findAll();
?>

10. Singleton Pattern

What is Singleton?

The Singleton pattern ensures that a class has only one instance and provides a global point of access to it.

Implementation

<?php
class Database {
    private static ?Database $instance = null;
    private PDO $connection;
    
    private function __construct() {
        $this->connection = new PDO(
            'mysql:host=localhost;dbname=myapp',
            'root',
            'password'
        );
    }
    
    private function __clone() {}
    
    public function __wakeup() {
        throw new Exception("Cannot unserialize singleton");
    }
    
    public static function getInstance(): Database {
        if (self::$instance === null) {
            self::$instance = new self();
        }
        return self::$instance;
    }
    
    public function getConnection(): PDO {
        return $this->connection;
    }
    
    public function query(string $sql): array {
        $stmt = $this->connection->query($sql);
        return $stmt->fetchAll();
    }
}

$db1 = Database::getInstance();
$db2 = Database::getInstance();

var_dump($db1 === $db2);

class Config {
    private static ?Config $instance = null;
    private array $settings = [];
    
    private function __construct() {
        $this->settings = [
            'app_name' => 'My Application',
            'debug' => true,
            'timezone' => 'UTC'
        ];
    }
    
    public static function getInstance(): Config {
        if (self::$instance === null) {
            self::$instance = new self();
        }
        return self::$instance;
    }
    
    public function get(string $key, $default = null) {
        return $this->settings[$key] ?? $default;
    }
    
    public function set(string $key, $value): void {
        $this->settings[$key] = $value;
    }
}

$config = Config::getInstance();
echo $config->get('app_name');
?>
⚠️ Caution: Singletons can make testing difficult and introduce global state. Consider using dependency injection instead when possible.

11. Factory Pattern

What is Factory Pattern?

The Factory pattern provides an interface for creating objects without specifying their exact classes. It encapsulates object creation logic.

Simple Factory

<?php
interface PaymentMethod {
    public function process(float $amount): array;
}

class CreditCardPayment implements PaymentMethod {
    public function process(float $amount): array {
        return [
            'status' => 'success',
            'method' => 'credit_card',
            'amount' => $amount
        ];
    }
}

class PayPalPayment implements PaymentMethod {
    public function process(float $amount): array {
        return [
            'status' => 'success',
            'method' => 'paypal',
            'amount' => $amount
        ];
    }
}

class BankTransferPayment implements PaymentMethod {
    public function process(float $amount): array {
        return [
            'status' => 'pending',
            'method' => 'bank_transfer',
            'amount' => $amount
        ];
    }
}

class PaymentFactory {
    public static function create(string $type): PaymentMethod {
        return match($type) {
            'credit_card' => new CreditCardPayment(),
            'paypal' => new PayPalPayment(),
            'bank_transfer' => new BankTransferPayment(),
            default => throw new InvalidArgumentException("Unknown payment type: $type")
        };
    }
}

$payment = PaymentFactory::create('credit_card');
$result = $payment->process(100.00);
print_r($result);

$payment = PaymentFactory::create('paypal');
$result = $payment->process(50.00);
print_r($result);
?>

Factory Method Pattern

<?php
interface Logger {
    public function log(string $message): void;
}

class FileLogger implements Logger {
    private string $filename;
    
    public function __construct(string $filename) {
        $this->filename = $filename;
    }
    
    public function log(string $message): void {
        file_put_contents($this->filename, $message . "\n", FILE_APPEND);
    }
}

class DatabaseLogger implements Logger {
    private PDO $connection;
    
    public function __construct(PDO $connection) {
        $this->connection = $connection;
    }
    
    public function log(string $message): void {
        $stmt = $this->connection->prepare(
            "INSERT INTO logs (message, created_at) VALUES (?, NOW())"
        );
        $stmt->execute([$message]);
    }
}

abstract class Application {
    abstract protected function createLogger(): Logger;
    
    public function run(): void {
        $logger = $this->createLogger();
        $logger->log('Application started');
    }
}

class FileBasedApp extends Application {
    protected function createLogger(): Logger {
        return new FileLogger('/var/log/app.log');
    }
}

class DatabaseApp extends Application {
    protected function createLogger(): Logger {
        $pdo = new PDO('mysql:host=localhost;dbname=app', 'root', 'pass');
        return new DatabaseLogger($pdo);
    }
}

$app = new FileBasedApp();
$app->run();

$app = new DatabaseApp();
$app->run();
?>

12. Repository Pattern

What is Repository Pattern?

The Repository pattern mediates between the domain and data mapping layers, acting like an in-memory collection of domain objects. It provides a collection-like interface for accessing domain objects.

Implementation

<?php
interface UserRepositoryInterface {
    public function find(int $id): ?User;
    public function findAll(): array;
    public function findByEmail(string $email): ?User;
    public function save(User $user): void;
    public function delete(int $id): void;
}

class User {
    private ?int $id;
    private string $name;
    private string $email;
    
    public function __construct(?int $id, string $name, string $email) {
        $this->id = $id;
        $this->name = $name;
        $this->email = $email;
    }
    
    public function getId(): ?int {
        return $this->id;
    }
    
    public function getName(): string {
        return $this->name;
    }
    
    public function getEmail(): string {
        return $this->email;
    }
}

class DatabaseUserRepository implements UserRepositoryInterface {
    private PDO $connection;
    
    public function __construct(PDO $connection) {
        $this->connection = $connection;
    }
    
    public function find(int $id): ?User {
        $stmt = $this->connection->prepare('SELECT * FROM users WHERE id = ?');
        $stmt->execute([$id]);
        $data = $stmt->fetch();
        
        return $data ? new User($data['id'], $data['name'], $data['email']) : null;
    }
    
    public function findAll(): array {
        $stmt = $this->connection->query('SELECT * FROM users');
        $users = [];
        
        while ($data = $stmt->fetch()) {
            $users[] = new User($data['id'], $data['name'], $data['email']);
        }
        
        return $users;
    }
    
    public function findByEmail(string $email): ?User {
        $stmt = $this->connection->prepare('SELECT * FROM users WHERE email = ?');
        $stmt->execute([$email]);
        $data = $stmt->fetch();
        
        return $data ? new User($data['id'], $data['name'], $data['email']) : null;
    }
    
    public function save(User $user): void {
        if ($user->getId()) {
            $stmt = $this->connection->prepare(
                'UPDATE users SET name = ?, email = ? WHERE id = ?'
            );
            $stmt->execute([$user->getName(), $user->getEmail(), $user->getId()]);
        } else {
            $stmt = $this->connection->prepare(
                'INSERT INTO users (name, email) VALUES (?, ?)'
            );
            $stmt->execute([$user->getName(), $user->getEmail()]);
        }
    }
    
    public function delete(int $id): void {
        $stmt = $this->connection->prepare('DELETE FROM users WHERE id = ?');
        $stmt->execute([$id]);
    }
}

class InMemoryUserRepository implements UserRepositoryInterface {
    private array $users = [];
    private int $nextId = 1;
    
    public function find(int $id): ?User {
        return $this->users[$id] ?? null;
    }
    
    public function findAll(): array {
        return array_values($this->users);
    }
    
    public function findByEmail(string $email): ?User {
        foreach ($this->users as $user) {
            if ($user->getEmail() === $email) {
                return $user;
            }
        }
        return null;
    }
    
    public function save(User $user): void {
        if ($user->getId()) {
            $this->users[$user->getId()] = $user;
        } else {
            $id = $this->nextId++;
            $this->users[$id] = new User($id, $user->getName(), $user->getEmail());
        }
    }
    
    public function delete(int $id): void {
        unset($this->users[$id]);
    }
}

?>

13. Observer Pattern

What is Observer Pattern?

The Observer pattern defines a one-to-many dependency between objects so that when one object changes state, all its dependents are notified and updated automatically.

Implementation

<?php
interface Observer {
    public function update(string $event, array $data): void;
}

interface Subject {
    public function attach(Observer $observer): void;
    public function detach(Observer $observer): void;
    public function notify(string $event, array $data): void;
}

class User implements Subject {
    private array $observers = [];
    private string $name;
    private string $email;
    
    public function __construct(string $name, string $email) {
        $this->name = $name;
        $this->email = $email;
    }
    
    public function attach(Observer $observer): void {
        $this->observers[] = $observer;
    }
    
    public function detach(Observer $observer): void {
        $key = array_search($observer, $this->observers, true);
        if ($key !== false) {
            unset($this->observers[$key]);
        }
    }
    
    public function notify(string $event, array $data): void {
        foreach ($this->observers as $observer) {
            $observer->update($event, $data);
        }
    }
    
    public function updateEmail(string $newEmail): void {
        $oldEmail = $this->email;
        $this->email = $newEmail;
        
        $this->notify('email_changed', [
            'user' => $this->name,
            'old_email' => $oldEmail,
            'new_email' => $newEmail
        ]);
    }
    
    public function register(): void {
        $this->notify('user_registered', [
            'name' => $this->name,
            'email' => $this->email
        ]);
    }
}

class EmailNotifier implements Observer {
    public function update(string $event, array $data): void {
        if ($event === 'user_registered') {
            echo "Sending welcome email to {$data['email']}\n";
        } elseif ($event === 'email_changed') {
            echo "Sending confirmation email to {$data['new_email']}\n";
        }
    }
}

class LoggerObserver implements Observer {
    public function update(string $event, array $data): void {
        $message = date('Y-m-d H:i:s') . " - Event: $event - Data: " . json_encode($data);
        echo "Logging: $message\n";
    }
}

class SlackNotifier implements Observer {
    public function update(string $event, array $data): void {
        if ($event === 'user_registered') {
            echo "Slack: New user registered - {$data['name']}\n";
        }
    }
}

$user = new User('John Doe', 'john@example.com');

$user->attach(new EmailNotifier());
$user->attach(new LoggerObserver());
$user->attach(new SlackNotifier());

$user->register();
$user->updateEmail('newemail@example.com');
?>

14. Abstract Classes vs Interfaces

Key Differences

Feature Abstract Class Interface
Methods Can have abstract and concrete methods Only method signatures (PHP 8+ allows default implementations)
Properties Can have properties Cannot have properties (only constants)
Inheritance Single inheritance only Multiple implementation allowed
Access Modifiers Can use public, protected, private All methods are public
Constructor Can have constructor Cannot have constructor
Purpose Share code among related classes Define a contract for unrelated classes

When to Use Abstract Classes

<?php
abstract class Vehicle {
    protected string $brand;
    protected int $year;
    protected float $price;
    
    public function __construct(string $brand, int $year, float $price) {
        $this->brand = $brand;
        $this->year = $year;
        $this->price = $price;
    }
    
    abstract public function start(): void;
    abstract public function stop(): void;
    
    public function getInfo(): string {
        return "{$this->year} {$this->brand} - $" . number_format($this->price, 2);
    }
    
    public function calculateDepreciation(int $years): float {
        return $this->price * (0.85 ** $years);
    }
}

class Car extends Vehicle {
    private int $doors;
    
    public function __construct(string $brand, int $year, float $price, int $doors) {
        parent::__construct($brand, $year, $price);
        $this->doors = $doors;
    }
    
    public function start(): void {
        echo "Car engine started\n";
    }
    
    public function stop(): void {
        echo "Car engine stopped\n";
    }
}

class Motorcycle extends Vehicle {
    private string $type;
    
    public function __construct(string $brand, int $year, float $price, string $type) {
        parent::__construct($brand, $year, $price);
        $this->type = $type;
    }
    
    public function start(): void {
        echo "Motorcycle engine started\n";
    }
    
    public function stop(): void {
        echo "Motorcycle engine stopped\n";
    }
}
?>

15. Method Overloading & Overriding

Method Overriding

Method overriding occurs when a child class provides a specific implementation of a method that is already defined in its parent class.

<?php
class Animal {
    protected string $name;
    
    public function __construct(string $name) {
        $this->name = $name;
    }
    
    public function makeSound(): string {
        return "Some generic sound";
    }
    
    public function getInfo(): string {
        return "Animal: {$this->name}";
    }
}

class Dog extends Animal {
    private string $breed;
    
    public function __construct(string $name, string $breed) {
        parent::__construct($name);
        $this->breed = $breed;
    }
    
    public function makeSound(): string {
        return "Woof! Woof!";
    }
    
    public function getInfo(): string {
        return parent::getInfo() . ", Breed: {$this->breed}";
    }
}

class Cat extends Animal {
    public function makeSound(): string {
        return "Meow!";
    }
}

$dog = new Dog("Buddy", "Golden Retriever");
echo $dog->makeSound();
echo $dog->getInfo();

$cat = new Cat("Whiskers");
echo $cat->makeSound();
?>

Method Overloading (Magic Methods)

PHP doesn't support traditional method overloading, but you can simulate it using magic methods like __call():

<?php
class Calculator {
    public function __call(string $method, array $arguments) {
        if ($method === 'add') {
            return $this->handleAdd($arguments);
        }
        
        throw new BadMethodCallException("Method $method does not exist");
    }
    
    private function handleAdd(array $arguments): float {
        $count = count($arguments);
        
        if ($count === 2) {
            return $arguments[0] + $arguments[1];
        } elseif ($count === 3) {
            return $arguments[0] + $arguments[1] + $arguments[2];
        }
        
        return array_sum($arguments);
    }
}

$calc = new Calculator();
echo $calc->add(5, 3);
echo $calc->add(5, 3, 2);
echo $calc->add(1, 2, 3, 4, 5);

class FlexibleMethod {
    public function process(...$args) {
        $count = count($args);
        
        if ($count === 1 && is_string($args[0])) {
            return $this->processString($args[0]);
        } elseif ($count === 1 && is_array($args[0])) {
            return $this->processArray($args[0]);
        } elseif ($count === 2) {
            return $this->processTwoArgs($args[0], $args[1]);
        }
        
        throw new InvalidArgumentException("Invalid arguments");
    }
    
    private function processString(string $value): string {
        return strtoupper($value);
    }
    
    private function processArray(array $value): int {
        return count($value);
    }
    
    private function processTwoArgs($first, $second): string {
        return "$first and $second";
    }
}

$obj = new FlexibleMethod();
echo $obj->process("hello");
echo $obj->process([1, 2, 3]);
echo $obj->process("foo", "bar");
?>

17. Final Keyword

What is Final?

The final keyword prevents a class from being extended or a method from being overridden.

Final Classes

<?php
final class SecurityManager {
    private string $encryptionKey;
    
    public function __construct(string $encryptionKey) {
        $this->encryptionKey = $encryptionKey;
    }
    
    public function encrypt(string $data): string {
        return base64_encode($data);
    }
    
    public function decrypt(string $data): string {
        return base64_decode($data);
    }
}

final class ImmutableValue {
    private $value;
    
    public function __construct($value) {
        $this->value = $value;
    }
    
    public function getValue() {
        return $this->value;
    }
}
?>

Final Methods

<?php
class BaseController {
    final public function execute(): void {
        $this->beforeAction();
        $this->action();
        $this->afterAction();
    }
    
    protected function beforeAction(): void {
        echo "Before action\n";
    }
    
    protected function action(): void {
        echo "Main action\n";
    }
    
    protected function afterAction(): void {
        echo "After action\n";
    }
}

class UserController extends BaseController {
    protected function action(): void {
        echo "User-specific action\n";
    }
}

$controller = new UserController();
$controller->execute();

class PaymentProcessor {
    final public function processPayment(float $amount): array {
        $this->validateAmount($amount);
        $this->logTransaction($amount);
        
        return $this->executePayment($amount);
    }
    
    protected function validateAmount(float $amount): void {
        if ($amount <= 0) {
            throw new InvalidArgumentException("Amount must be positive");
        }
    }
    
    protected function logTransaction(float $amount): void {
        echo "Processing payment of $$amount\n";
    }
    
    protected function executePayment(float $amount): array {

    }

    }
}

💡 When to Use Final:
  • Security-critical classes that shouldn't be extended
  • Methods that must maintain specific behavior
  • Template methods that define a workflow
  • Value objects that should remain immutable

18. Dependency Injection

What is Dependency Injection?

Dependency Injection (DI) is a design pattern where an object receives its dependencies from external sources rather than creating them itself. This promotes loose coupling and easier testing.

Constructor Injection

<?php
interface LoggerInterface {
    public function log(string $message): void;
}

class FileLogger implements LoggerInterface {
    private string $filename;
    
    public function __construct(string $filename) {
        $this->filename = $filename;
    }
    
    public function log(string $message): void {
        file_put_contents($this->filename, $message . "\n", FILE_APPEND);
    }
}

interface EmailServiceInterface {
    public function send(string $to, string $subject, string $body): bool;
}

class SmtpEmailService implements EmailServiceInterface {
    private string $smtpHost;
    
    public function __construct(string $smtpHost) {
        $this->smtpHost = $smtpHost;
    }
    
    public function send(string $to, string $subject, string $body): bool {
        echo "Sending email via {$this->smtpHost} to $to\n";
        return true;
    }
}

class UserService {
    private UserRepositoryInterface $repository;
    private LoggerInterface $logger;
    private EmailServiceInterface $emailService;
    
    public function __construct(
        UserRepositoryInterface $repository,
        LoggerInterface $logger,
        EmailServiceInterface $emailService

        $this->repository = $repository;

Service Container (DI Container)

<?php
class Container {
    private array $bindings = [];
    private array $instances = [];
    
    public function bind(string $abstract, callable $concrete): void {
        $this->bindings[$abstract] = $concrete;
    }
    
    public function singleton(string $abstract, callable $concrete): void {
        $this->bindings[$abstract] = function() use ($abstract, $concrete) {
            if (!isset($this->instances[$abstract])) {
                $this->instances[$abstract] = $concrete($this);
            }
            return $this->instances[$abstract];
        };
    }
    
    public function make(string $abstract) {
        if (isset($this->bindings[$abstract])) {
            return $this->bindings[$abstract]($this);
        }
        
        throw new Exception("No binding found for $abstract");
    }
}

$container = new Container();

$container->bind(LoggerInterface::class, function($container) {
    return new FileLogger('/var/log/app.log');
});

$container->bind(EmailServiceInterface::class, function($container) {
    return new SmtpEmailService('smtp.example.com');
});

$container->singleton(DatabaseInterface::class, function($container) {
    return new MySQLDatabase('localhost', 'myapp');
});

$container->bind(UserRepositoryInterface::class, function($container) {
    return new DatabaseUserRepository(
        $container->make(DatabaseInterface::class)
    );
});

$container->bind(UserService::class, function($container) {
    return new UserService(
        $container->make(UserRepositoryInterface::class),
        $container->make(LoggerInterface::class),
        $container->make(EmailServiceInterface::class)
    );
});

$userService = $container->make(UserService::class);
?>
