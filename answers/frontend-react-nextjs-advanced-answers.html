1. React 18 Concurrent Features

Automatic Batching

React 18 automatically batches state updates in event handlers, timeouts, promises, and async functions to reduce re-renders.

// Before React 18: Multiple re-renders
function handleClick() {
  setCount(c => c + 1); // Re-render
  setFlag(f => !f);     // Re-render
  setItems([...items]); // Re-render
}

// React 18: Single re-render (automatic batching)
function handleClick() {
  setCount(c => c + 1);
  setFlag(f => !f);
  setItems([...items]);
  // All batched into one re-render
}

// Car booking example
function CarBooking() {
  const [selectedCar, setSelectedCar] = useState(null);
  const [bookingDate, setBookingDate] = useState(null);
  const [isLoading, setIsLoading] = useState(false);

  async function handleBooking(carId, date) {
    const response = await fetch('/api/book');
    const data = await response.json();
    
    // All updates batched automatically
    setSelectedCar(data.car);
    setBookingDate(data.date);
    setIsLoading(false);
  }

  return /* JSX */;
}

Opting Out of Automatic Batching

import { flushSync } from 'react-dom';

function handleClick() {
  flushSync(() => {
    setCount(c => c + 1);
  }); // Re-render happens immediately
  
  flushSync(() => {
    setFlag(f => !f);
  }); // Another re-render
}

useTransition Hook

Mark state updates as non-urgent transitions, allowing React to keep the UI responsive during expensive renders.

import { useState, useTransition } from 'react';

function CarFilter() {
  const [searchTerm, setSearchTerm] = useState('');
  const [filteredCars, setFilteredCars] = useState(cars);
  const [isPending, startTransition] = useTransition();

  function handleSearch(e) {
    const value = e.target.value;
    setSearchTerm(value); // Urgent: update input immediately
    
    startTransition(() => {
      // Non-urgent: filter can be delayed
      const filtered = cars.filter(car =>
        car.name.toLowerCase().includes(value.toLowerCase())
      );
      setFilteredCars(filtered);
    });
  }

  return (
    <div>
      <input 
        value={searchTerm} 
        onChange={handleSearch}
        placeholder="Search cars..."
      />
      {isPending && <Spinner />}
      <CarList cars={filteredCars} />
    </div>
  );
}

useDeferredValue Hook

Defer updating part of the UI to keep it responsive during expensive renders.

import { useState, useDeferredValue, useMemo } from 'react';

function CarSearch() {
  const [searchTerm, setSearchTerm] = useState('');
  const deferredSearchTerm = useDeferredValue(searchTerm);

  const filteredCars = useMemo(() => {
    return cars.filter(car =>
      car.name.toLowerCase().includes(deferredSearchTerm.toLowerCase())
    );
  }, [deferredSearchTerm]);

  const isStale = searchTerm !== deferredSearchTerm;

  return (
    <div>
      <input 
        value={searchTerm}
        onChange={e => setSearchTerm(e.target.value)}
      />
      <div style={{ opacity: isStale ? 0.5 : 1 }}>
        <CarList cars={filteredCars} />
      </div>
    </div>
  );
}

useSyncExternalStore Hook

Subscribe to external stores safely with concurrent rendering.

import { useSyncExternalStore } from 'react';

function useCarStore() {
  return useSyncExternalStore(
    carStore.subscribe,
    carStore.getSnapshot,
    carStore.getServerSnapshot
  );
}

// External store implementation
const carStore = {
  state: { cars: [], selectedCar: null },
  listeners: new Set(),
  
  subscribe(listener) {
    this.listeners.add(listener);
    return () => this.listeners.delete(listener);
  },
  
  getSnapshot() {
    return this.state;
  },
  
  getServerSnapshot() {
    return this.state;
  },
  
  setState(newState) {
    this.state = { ...this.state, ...newState };
    this.listeners.forEach(listener => listener());
  }
};

'use server';

import { revalidatePath } from 'next/cache';
import { db } from '@/lib/db';

export async function bookCar(formData: FormData) {
  const carId = formData.get('carId') as string;
  const userId = formData.get('userId') as string;
  const startDate = formData.get('startDate') as string;
  const endDate = formData.get('endDate') as string;

  revalidatePath(`/bookings/${booking.id}`);

  return { success: true, bookingId: booking.id };
}

// middleware.ts
import { NextResponse } from 'next/server';
import type { NextRequest } from 'next/server';

export function middleware(request: NextRequest) {
  const token = request.cookies.get('token');
  
  // Redirect to login if not authenticated
  if (!token && request.nextUrl.pathname.startsWith('/dashboard')) {
    return NextResponse.redirect(new URL('/login', request.url));
  }
  
  // Add custom header
  const response = NextResponse.next();
  response.headers.set('x-custom-header', 'value');
  
  return response;
}

import { memo, useMemo } from 'react';

// Memoize expensive computation
function CarCatalog({ cars, filters }) {
  const filteredAndSortedCars = useMemo(() => {
    return cars
      .filter(car => {
        return (!filters.brand || car.brandId === filters.brand) &&
               (!filters.maxPrice || car.price <= filters.maxPrice);
      })
      .sort((a, b) => {
        if (filters.sortBy === 'price') return a.price - b.price;
        if (filters.sortBy === 'name') return a.name.localeCompare(b.name);
        return 0;
      });
  }, [cars, filters]);

  return (
    <div>
      {filteredAndSortedCars.map(car => (
        <CarCard key={car.id} car={car} />
      ))}
    </div>
  );
}

4. State Management (Zustand/Redux)

Zustand - Lightweight State Management

Simple, unopinionated state management with minimal boilerplate.

// stores/carStore.ts
import { create } from 'zustand';
import { devtools, persist } from 'zustand/middleware';

interface Car {
  id: string;
  name: string;
  price: number;
  brandId: string;
}

interface CarStore {
  cars: Car[];
  selectedCar: Car | null;
  isLoading: boolean;
  error: string | null;
  fetchCars: () => Promise<void>;
  selectCar: (carId: string) => void;
  clearSelection: () => void;
}

export const useCarStore = create<CarStore>()(
  devtools(
    persist(
      (set, get) => ({
        cars: [],
        selectedCar: null,
        isLoading: false,
        error: null,
        
        fetchCars: async () => {
          set({ isLoading: true, error: null });
          try {
            const response = await fetch('/api/cars');
            const cars = await response.json();
            set({ cars, isLoading: false });
          } catch (error) {
            set({ error: error.message, isLoading: false });
          }
        },
        
        selectCar: (carId) => {
          const car = get().cars.find(c => c.id === carId);
          set({ selectedCar: car || null });
        },
        
        clearSelection: () => set({ selectedCar: null })
      }),
      { name: 'car-storage' }
    )
  )
);

// Component usage
function CarCatalog() {
  const { cars, isLoading, fetchCars } = useCarStore();
  
  useEffect(() => {
    fetchCars();
  }, [fetchCars]);
  
  if (isLoading) return <Spinner />;
  
  return (
    <div>
      {cars.map(car => <CarCard key={car.id} car={car} />)}
    </div>
  );
}

// Selecting specific state
function CarDetails() {
  const selectedCar = useCarStore(state => state.selectedCar);
  const clearSelection = useCarStore(state => state.clearSelection);
  
  if (!selectedCar) return null;
  
  return (
    <div>
      <h2>{selectedCar.name}</h2>
      <button onClick={clearSelection}>Close</button>
    </div>
  );
}

function useBookCar() {
  const queryClient = useQueryClient();
  
  return useMutation({
    mutationFn: async (bookingData) => {
      const response = await fetch('/api/bookings', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(bookingData)
      });
      return response.json();
    },
    onMutate: async (newBooking) => {
      await queryClient.cancelQueries({ queryKey: ['cars', newBooking.carId] });
      const previousCar = queryClient.getQueryData(['cars', newBooking.carId]);
      
      queryClient.setQueryData(['cars', newBooking.carId], (old) => ({
        ...old,
        available: false
      }));
      
      return { previousCar };
    },
    onError: (err, newBooking, context) => {
      queryClient.setQueryData(
        ['cars', newBooking.carId],
        context.previousCar
      );
    },
    onSettled: (data, error, variables) => {
      queryClient.invalidateQueries({ queryKey: ['cars'] });
      queryClient.invalidateQueries({ queryKey: ['bookings'] });
    }
  });
}

useLocalStorage Hook

import { useState, useEffect } from 'react';

function useLocalStorage<T>(key: string, initialValue: T) {
  const [storedValue, setStoredValue] = useState<T>(() => {
    if (typeof window === 'undefined') {
      return initialValue;
    }
    try {
      const item = window.localStorage.getItem(key);
      return item ? JSON.parse(item) : initialValue;
    } catch (error) {
      console.error(error);
      return initialValue;
    }
  });

  const setValue = (value: T | ((val: T) => T)) => {
    try {
      const valueToStore = value instanceof Function ? value(storedValue) : value;
      setStoredValue(valueToStore);
      if (typeof window !== 'undefined') {
        window.localStorage.setItem(key, JSON.stringify(valueToStore));
      }
    } catch (error) {
      console.error(error);
    }
  };

  return [storedValue, setValue] as const;
}

useDebounce Hook

import { useState, useEffect } from 'react';

function useDebounce<T>(value: T, delay: number): T {
  const [debouncedValue, setDebouncedValue] = useState<T>(value);

  useEffect(() => {
    const handler = setTimeout(() => {
      setDebouncedValue(value);
    }, delay);

    return () => {
      clearTimeout(handler);
    };
  }, [value, delay]);

  return debouncedValue;
}

// Usage
function CarSearch() {

}

useIntersectionObserver Hook

import { useEffect, useRef, useState } from 'react';

function useIntersectionObserver(
  options?: IntersectionObserverInit
) {
  const [isIntersecting, setIsIntersecting] = useState(false);
  const targetRef = useRef<HTMLDivElement>(null);

  useEffect(() => {
    const target = targetRef.current;
    if (!target) return;

    const observer = new IntersectionObserver(
      ([entry]) => setIsIntersecting(entry.isIntersecting),
      options
    );

    observer.observe(target);

    return () => {
      observer.disconnect();
    };
  }, [options]);

  return { targetRef, isIntersecting };
}

// Usage: Lazy load images
function CarImage({ src, alt }) {
  const { targetRef, isIntersecting } = useIntersectionObserver({
    threshold: 0.1
  });
  const [shouldLoad, setShouldLoad] = useState(false);

  useEffect(() => {
    if (isIntersecting) {
      setShouldLoad(true);
    }
  }, [isIntersecting]);

  return (
    <div ref={targetRef}>
      {shouldLoad ? (
        <img src={src} alt={alt} />
      ) : (
        <div className="placeholder" />
      )}
    </div>
  );
}

useMediaQuery Hook

import { useState, useEffect } from 'react';

function useMediaQuery(query: string): boolean {
  const [matches, setMatches] = useState(false);

  useEffect(() => {
    const media = window.matchMedia(query);
    
    if (media.matches !== matches) {
      setMatches(media.matches);
    }

    const listener = () => setMatches(media.matches);
    media.addEventListener('change', listener);

    return () => media.removeEventListener('change', listener);
  }, [matches, query]);

  return matches;
}
